```js
// Построчный diff (git-стиль) — чистая функция, без зависимостей.
// Алгоритм: классический LCS по строкам (динамика), затем восстановление
// последовательности сегментов. Возвращает массив сегментов в порядке нового
// текста с вкраплением удалённых строк на их позиции.
//
// Вход:  oldText, newText — строки (целые тексты ТЗ).
// Выход: [{ type, text }] где type ∈ 'equal' | 'del' | 'add', text — одна строка.
//   'equal' — строка есть в обоих; 'del' — была в old, удалена; 'add' — добавлена в new.

function splitLines(s) {
  // Нормализуем переводы строк, не теряем пустые строки в середине.
  if (s == null) return [];
  const norm = String(s).replace(/\r\n/g, "\n").replace(/\r/g, "\n");
  // Пустой текст -> ноль строк (нет смысла в одной пустой строке-сегменте).
  if (norm === "") return [];
  return norm.split("\n");
}

export function diffLines(oldText, newText) {
  const a = splitLines(oldText);
  const b = splitLines(newText);
  const n = a.length, m = b.length;

  // LCS-таблица длин. (n+1)x(m+1).
  const dp = Array.from({ length: n + 1 }, () => new Array(m + 1).fill(0));
  for (let i = n - 1; i >= 0; i--) {
    for (let j = m - 1; j >= 0; j--) {
      dp[i][j] = a[i] === b[j]
        ? dp[i + 1][j + 1] + 1
        : Math.max(dp[i + 1][j], dp[i][j + 1]);
    }
  }

  // Восстановление: идём с начала, формируем сегменты в естественном порядке.
  const out = [];
  let i = 0, j = 0;
  while (i < n && j < m) {
    if (a[i] === b[j]) {
      out.push({ type: "equal", text: a[i] });
      i++; j++;
    } else if (dp[i + 1][j] >= dp[i][j + 1]) {
      out.push({ type: "del", text: a[i] });
      i++;
    } else {
      out.push({ type: "add", text: b[j] });
      j++;
    }
  }
  while (i < n) { out.push({ type: "del", text: a[i] }); i++; }
  while (j < m) { out.push({ type: "add", text: b[j] }); j++; }
  return out;
}
```

function DiffView({ oldText, newText }) {
  const segs = diffLines(oldText, newText);
  if (!segs.length) return <div className="text-xs opacity-50 py-1">— пусто —</div>;
  return (
    <div className="font-mono text-xs rounded bg-zinc-950 p-2 overflow-x-auto" style={{ whiteSpace: "pre-wrap", wordBreak: "break-word" }}>
      {segs.map((s, i) => {
        if (s.type === "del") return <div key={i} style={{ color: "#f8a3a3", background: "rgba(248,163,163,0.08)" }}>- {s.text || " "}</div>;
        if (s.type === "add") return <div key={i} style={{ color: "#6ee7a8", background: "rgba(110,231,168,0.08)" }}>+ {s.text || " "}</div>;
        return <div key={i} style={{ color: "#a8a8a3" }}>&nbsp;&nbsp;{s.text || " "}</div>;
      })}
    </div>
  );
}

## Task 7: Realtime-подписка списка/доски + точечный refetch открытой карточки

Один канал `project_tasks` через `postgres_changes`, мёрж в `tasks` TasksView, отписка при unmount. Клиентский фолбэк-фильтр приватности: новые/обновлённые строки принимаем в состояние, только если они проходят базовую проверку доступа на клиенте (см. ниже) — это страховка на случай, если Realtime отдаст лишнее (основная гарантия — RLS на канале, проверяется вручную в Task 8). Внутри открытой карточки — refetch версий/комментариев при событии по id открытой задачи.

**Files:**
- Modify: `src/App.jsx` — `TasksView` (~3101), проброс сигнала в `TaskModal`

- [ ] **Step 1: Добавить Realtime-эффект и tick открытой задачи в `TasksView` (после `useEffect(() => { reload(); }, [reload]);`, ~3121)**

```jsx
  // ── 6.4b: Realtime-подписка на project_tasks ──
  // tick инкрементируется при realtime-событии по id открытой задачи -> модалка
  // делает refetch версий/комментариев (дочерние таблицы не в publication).
  const [openTaskTick, setOpenTaskTick] = useState(0);

  // клиентский фолбэк-фильтр приватности: пускаем строку в состояние только если
  // она проходит базовую проверку доступа (страховка поверх RLS на канале).
  const canSeeRow = useCallback((row) => {
    if (!row) return false;
    if (profile?.role === "admin") return true;
    if (row.project_id == null) {
      // личная задача: видит автор или исполнитель
      return row.author_id === profile.id || row.assigned_to === profile.id;
    }
    // проектная: видна, если проект уже в доступном списке (projects проп) ИЛИ
    // пользователь — сторона задачи. Жёстче серверной RLS, но безопасно (не «течёт»).
    return projects.some(p => p.id === row.project_id) || row.author_id === profile.id || row.assigned_to === profile.id;
  }, [profile, projects]);

  useEffect(() => {
    const channel = client
      .channel("project_tasks")
      .on("postgres_changes", { event: "*", schema: "public", table: "project_tasks" }, (payload) => {
        const { eventType, new: newRow, old: oldRow } = payload;
        setTasks(prev => {
          if (eventType === "DELETE") return prev.filter(t => t.id !== oldRow.id);
          if (!canSeeRow(newRow)) {
            // строка вне доступа — убираем, если вдруг была в состоянии
            return prev.filter(t => t.id !== newRow.id);
          }
          const mapped = taskDbToJs(newRow); // payload не содержит *_name -> projectName/assigneeName будут null
          // сохраняем уже известные denormalized-имена из текущего состояния (payload их не несёт)
          const existing = prev.find(t => t.id === mapped.id);
          const merged = existing
            ? { ...mapped, projectName: mapped.projectName ?? existing.projectName, assigneeName: mapped.assigneeName ?? existing.assigneeName, authorName: mapped.authorName ?? existing.authorName }
            : mapped;
          const idx = prev.findIndex(t => t.id === merged.id);
          if (idx === -1) return [merged, ...prev];
          const copy = prev.slice(); copy[idx] = merged; return copy;
        });
        // точечный сигнал открытой карточке
        const rid = (payload.new && payload.new.id) || (payload.old && payload.old.id);
        if (editing && editing.id && rid === editing.id) setOpenTaskTick(t => t + 1);
      })
      .subscribe();
    return () => { client.removeChannel(channel); };
  }, [client, canSeeRow, editing]);
```

### Места «ПРОВЕРИТЬ» (допущения о коде 6.4a — исполнитель обязан сверить)

1. **Создание версии №1 для НОВОЙ задачи.** План 2 для `isNew` оставляет старый textarea `description` и `createTask` (прямой insert в `project_tasks`). Спек §2/План 1 backf-ил покрывает только **существующие** задачи. **ПРОВЕРИТЬ:** создаёт ли План 1 версию №1 approved для **вновь** создаваемых задач (триггер на insert `project_tasks` или внутри RPC создания). Если НЕТ — у новой задачи не будет «действующей approved-версии», и первое «Изменить ТЗ» сделает diff против `task.description` (фолбэк `currentTz = task.description` это покрывает), но в «Истории ТЗ» не будет v1. Если требуется v1 для новых задач — согласовать с владельцем (либо триггер в Плане 1, либо вызов `propose_tz_version` сразу после `createTask` — но он создаст pending, не approved). Текущий план не падает в любом случае (есть фолбэк `currentTz`).

2. **`profile.role === "admin"`.** Используется в `canResolveQ` (Task 6) и `canSeeRow` (Task 7) по образцу `CommentsSection` (`profile?.role === "admin"`). **ПРОВЕРИТЬ:** что в объекте `profile`, прокинутом в `TasksView`/`TaskModal`, реально есть поле `role` со значением `"admin"` для админа (в 6.4a `CommentsSection` это поле использует — высокая вероятность, но сверить на месте).

3. **`payload.new` колонки для `postgres_changes`.** Допущение: payload несёт «сырые» колонки строки `project_tasks` (`id, project_id, author_id, assigned_to, status, ...`) без вычисляемых `project_name`/`assignee_name` (они только в `get_tasks` RPC). Поэтому имена в мёрже берутся из текущего состояния. **ПРОВЕРИТЬ:** на self-hosted Supabase Realtime REPLICA IDENTITY таблицы `project_tasks` — для корректного `payload.old` при DELETE/UPDATE нужна `REPLICA IDENTITY FULL` (иначе `old` содержит только PK). Для нашей логики DELETE достаточно `old.id` (PK всегда есть) — критично только если потребуется больше полей из `old`. Если карточки при чужом UPDATE «теряют» имена надолго — это ожидаемо до следующего `reload()`; при необходимости заменить мёрж на точечный `reload()` в обработчике (проще, но дороже по запросам).

4. **Realtime RLS на self-hosted.** §5 спека прямо требует проверить, что `postgres_changes` уважает RLS. **ПРОВЕРИТЬ вручную (Task 8 Step 4):** под посторонним C приватная задача A не должна приходить. Если RLS на канале не включён на этом инстансе — BLOCKER, эскалация (настройка Realtime authorization), пока — фолбэк отсекает на клиенте.
